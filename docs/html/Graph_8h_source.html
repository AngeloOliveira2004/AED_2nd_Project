146template<class T>
147int Graph<T>::calculateDiameter() const {
148 int max_diameter = 0;
149 for(auto a : getVertexSet())
150 {
151 a->setVisited(false);
152 a->setParent(nullptr);
153 a->setNum(0);
154 }
155 for (auto a : getVertexSet()){
156 T airport = a->getInfo();
157 this->bfsDifferent(airport);
158 for (Vertex<T>* b : getVertexSet()){
159 if (b->getNum() > max_diameter)
160 {
161 max_diameter = b->getNum();
162 }
163 }
164 }
165
166 return max_diameter;
167}

657/*
658 * Performs a depth-first search (dfs) traversal in a graph (this).

709/****************** BFS ********************/
710/*
711 * Performs a breadth-first search (bfs) in a graph (this), starting
712 * from the vertex with the given source contents (source).
713 * Returns a vector with the contents of the vertices by bfs order.
714 */
725template <class T>
726vector<T> Graph<T>::bfs(const T & source) const {
727 vector<T> res;
728 auto s = findVertex(source);
729 if (s == NULL)
730 return res;
731 queue<Vertex<T> *> q;
732 for (auto v : vertexSet)
733 v->visited = false;
734 q.push(s);
735 s->visited = true;
736 while (!q.empty()) {
737 auto v = q.front();
738 q.pop();
739 res.push_back(v->info);
740 for (auto & e : v->adj) {
741 auto w = e.dest;
742 if ( ! w->visited ) {
743 q.push(w);
744 w->visited = true;
745 }
746 }
747 }
748}
758template<class T>
759void Graph<T>::bfsDifferent(T &source) const {
760 auto s = findVertex(source);
761 if (s == NULL)
762 return ;
763 queue<Vertex<T> *> q;
764 for (auto v : vertexSet)
765 v->setVisited(false);
766 q.push(s);
767 s->setVisited(true);
768 s->setNum(0);
769 while (!q.empty()) {
770 auto v = q.front();
771 q.pop();
772 for (const Edge<T> & e : v->getAdj()) {
773 Vertex<T>* w = e.getDest();
774 if (!w->isVisited()) {
775 q.push(w);
776 w->setVisited(true);
777 w->setNum(v->getNum() + 1);
778 }
779 }
780 }

783/****************** isDAG ********************/
784/*
785 * Performs a depth-first search in a graph (this), to determine if the graph
786 * is acyclic (acyclic directed graph or DAG).
787 * During the search, a cycle is found if an edge connects to a vertex
788 * that is being processed in the stack of recursive calls (see theoretical classes).
789 * Returns true if the graph is acyclic, and false otherwise.
790 */
791
792template <class T>
793bool Graph<T>::isDAG() const {
794 for (auto v : vertexSet) {
795 v->visited = false;
796 v->processing = false;
797 }
798 for (auto v : vertexSet)
799 if (! v->visited)
800 if ( ! dfsIsDAG(v) )
801 return false;
802 return true;
803}
804
809template <class T>
810bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
811 v->visited = true;
812 v->processing = true;
813 for (auto & e : v->adj) {
814 auto w = e.dest;
815 if (w->processing)
816 return false;
817 if (! w->visited)
818 if (! dfsIsDAG(w))
819 return false;
820 }
821 v->processing = false;
822 return true;
823}

828/*
829 * Performs a topological sorting of the vertices of a graph (this).
830 * Returns a vector with the contents of the vertices by topological order.
831 * If the graph has cycles, returns an empty vector.
832 * Follows the algorithm described in theoretical classes.
833 */
844template<class T>
845void topsortAux(Vertex<T>* vertex, stack<Vertex<T>*> & s) {
846 vertex->setVisited(true);
847 for (Edge<T> edge : vertex->getAdj()) {
848 auto neighbor = edge.getDest();
849 if (!neighbor->isVisited()) {
850 topsortAux(neighbor, s);
851 }
852 }
853 s.push(vertex);
854}

866template<class T>
867vector<T> Graph<T>::topsort() const {
868 vector<T> res;
869 stack<Vertex<T>*> s;
870 for (Vertex<T> * vertex : this->vertexSet) {
871 vertex->setVisited(false);
872 }
873 for (Vertex<T>* vertex : this->vertexSet) {
874 if (!vertex->isVisited())
875 topsortAux(vertex, s);
876 }
877
878 while (!s.empty()) {
879 res.push_back(s.top()->getInfo());
880 s.pop();
881 }
882 return res;
883}
